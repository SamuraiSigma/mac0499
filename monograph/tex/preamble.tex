% Language and encoding
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage[brazil]{babel}

% Fonts: normal and monospaced
\usepackage{mathpazo}
\usepackage{inconsolata}

% Additional commands to math mode
\usepackage{amsmath}

% Checmark symbol (\checkmark)
\usepackage{amssymb}

% Indent first line of first paragraph
\usepackage{indentfirst}

% Allow access date, author, title

% Don't put page number on empty page
\usepackage{emptypage}

% Line breaks URLs on '-' and '_'
\usepackage[hyphens]{url}

% Remove space after a comma when it acts like a decimal separator
\usepackage{icomma}

\usepackage[usenames,svgnames,dvipsnames]{xcolor}

\usepackage{enumerate}

\usepackage{multicol}

\usepackage[pdftex]{graphicx}

% Input raw text
\usepackage{verbatim}

% Include external raw PDF pages
\usepackage{pdfpages}

% Allow forcing a figure to be displayed "here"

% Link colors
\usepackage[pdftex,
            plainpages=false,
            pdfpagelabels,
            pagebackref,
            colorlinks=true,
            citecolor=DarkGreen,
            linkcolor=NavyBlue,
            urlcolor=DarkRed,
